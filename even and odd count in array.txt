package practice;
import java.util.Scanner;
public class saveetha
{
public static void main(String[] args)
{
Scanner input = new Scanner(System.in);
int n,ec=0,oc=0,i;
n=input.nextInt();
int arr[]=new int[n];
for(i=0;i<n;i++)
	arr[i]=input.nextInt();
for(i=0;i<n;i++)
{
if(arr[i]%2==0)
{
	ec++;
}
else
{
	oc++;
}
}
System.out.println("even count is:"+ec);
System.out.println("odd count is:"+oc);
}
}
