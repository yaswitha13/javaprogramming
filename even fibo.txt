package studyopedia;

public class sumFibo {

	public static void main(String[] args) {
		int n=4,first=0,second=1,nt,c=0;
		System.out.println(first);
		System.out.println(second);
		while(n != 0)
		{
			nt=first+second;
			
			
			System.out.println(nt);
			if(nt % 2 ==0)
				c=c+nt;
			first=second;
			second=nt;
			n--;
		}
		System.out.println("even count is:"+c);

	}

}
